// server/ai/callAI.js
// Minimal OpenAI call using fetch (works on Node 18+). Returns parsed JSON.
export async function callAI(prompt) {
  const key = process.env.OPENAI_API_KEY;
  if (!key) throw new Error("OPENAI_API_KEY not set. Add it in Replit Secrets.");

  const resp = await fetch("https://api.openai.com/v1/chat/completions", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${key}`,
    },
    body: JSON.stringify({
      model: "gpt-4o-mini",            // small + cheap; change if you like
      temperature: 0.2,
      messages: [
        { role: "system", content: "You are a nursing board exam item writer." },
        { role: "user", content: prompt },
      ],
    }),
  });

  if (!resp.ok) {
    const text = await resp.text();
    throw new Error(`AI HTTP ${resp.status}: ${text}`);
  }

  const data = await resp.json();
  const content = data?.choices?.[0]?.message?.content || "";

  // Extract JSON block from the response (robust to extra text)
  const jsonMatch = content.match(/\{[\s\S]*\}$/);
  const jsonText = jsonMatch ? jsonMatch[0] : content;

  let parsed;
  try {
    parsed = JSON.parse(jsonText);
  } catch (e) {
    throw new Error("AI did not return valid JSON");
  }
  return parsed;
}

C) server/routes/aiQuiz.js
// server/routes/aiQuiz.js
import express from "express";
import { pool } from "../db.js";
import { callAI } from "../ai/callAI.js";

const router = express.Router();

// Helper: validate/normalize one item
function normalizeItem(it) {
  if (!it || typeof it !== "object") return null;
  const { question, options, correct_index, difficulty, rationale, tags } = it;

  if (!question || !Array.isArray(options) || options.length !== 4) return null;
  if (typeof correct_index !== "number" || correct_index < 0 || correct_index > 3) return null;

  return {
    question: String(question).trim(),
    options,
    correct_index,
    difficulty: ["easy", "medium", "hard"].includes(difficulty) ? difficulty : "medium",
    rationale: rationale ? String(rationale) : null,
    tags: Array.isArray(tags) ? tags.map(t => String(t).toLowerCase()) : [],
  };
}

/**
 * POST /ai-quiz/generate
 * body: { subject_code: "NCM6103", n?:10, difficulty?: "mixed" }
 * Creates n AI questions scoped to the subject's topic tags, stores in quiz_items, and returns count.
 */
router.post("/generate", async (req, res) => {
  const { subject_code, n = 10, difficulty = "mixed" } = req.body || {};
  if (!subject_code) return res.status(400).json({ error: "subject_code is required" });

  try {
    // 1) Pull tags from DB
    const top = await pool.query(
      "SELECT topic_name, tags FROM subject_topics WHERE subject_code = $1",
      [subject_code]
    );
    if (top.rowCount === 0) {
      return res.status(404).json({ error: "No topics/tags found for subject_code" });
    }
    const tagPool = [...new Set(top.rows.flatMap(r => r.tags || []))].slice(0, 30);

    // 2) Difficulty blueprint
    const mix =
      difficulty === "easy" ? { easy: 1, medium: 0, hard: 0 } :
      difficulty === "hard" ? { easy: 0, medium: 0, hard: 1 } :
      { easy: 0.4, medium: 0.4, hard: 0.2 }; // "mixed"

    // 3) Build the prompt
    const prompt = `
Create ${n} multiple-choice nursing questions for subject "${subject_code}" ONLY using these topic tags:
${JSON.stringify(tagPool)}.

Blueprint:
- Difficulty mix: ${JSON.stringify(mix)} (label each item: "easy" | "medium" | "hard")
- 4 options exactly per item (A–D). One correct answer (use 0..3 index).
- PNLE/NCLEX style; clinical reasoning; clear stems; no trick questions.
- Keep context Filipino-friendly where appropriate.
- Provide a brief rationale.

Return STRICT JSON ONLY in this shape:
{
  "items": [
    {
      "question": "text",
      "options": ["A","B","C","D"],
      "correct_index": 0,
      "difficulty": "easy|medium|hard",
      "rationale": "why correct & why others are not",
      "tags": ["one or more from provided list"]
    }
  ]
}`;

    // 4) Call AI
    const ai = await callAI(prompt);
    if (!ai?.items) throw new Error("AI returned no items");

    // 5) Store validated items
    let created = 0;
    for (const raw of ai.items) {
      const item = normalizeItem(raw);
      if (!item) continue;

      // basic duplicate guard by stem
      const exists = await pool.query(
        "SELECT 1 FROM quiz_items WHERE subject_code = $1 AND question = $2 LIMIT 1",
        [subject_code, item.question]
      );
      if (exists.rowCount) continue;

      await pool.query(
        `INSERT INTO quiz_items
         (subject_code, topic_name, question, choices, correct_index, difficulty, tags, rationale, source_note)
         VALUES ($1,$2,$3,$4,$5,$6,$7,$8,$9)`,
        [
          subject_code,
          null,
          item.question,
          JSON.stringify(item.options),
          item.correct_index,
          item.difficulty,
          item.tags,
          item.rationale,
          "ai_gen_v1",
        ]
      );
      created++;
    }

    res.json({ ok: true, created });
  } catch (e) {
    console.error(e);
    res.status(500).json({ error: e.message });
  }
});

/**
 * GET /quiz/pull?subject_code=NCM6103&n=10
 * Returns stored questions WITHOUT the correct answer (anti-cheat).
 */
router.get("/pull", async (req, res) => {
  const subject = String(req.query.subject_code || "");
  const n = Math.min(parseInt(req.query.n || "10", 10), 100);
  if (!subject) return res.status(400).json({ error: "subject_code required" });

  const { rows } = await pool.query(
    `SELECT id, question, choices, difficulty
     FROM quiz_items
     WHERE subject_code = $1
     ORDER BY RANDOM()
     LIMIT $2`,
    [subject, n]
  );

  const items = rows.map(r => ({
    id: r.id,
    question: r.question,
    options: r.choices,
    difficulty: r.difficulty,
  }));

  res.json({ items });
});

/**
 * POST /quiz/submit
 * body: { answers: [{ id, choiceIndex }] }
 * Returns score and per-item correctness.
 */
router.post("/submit", async (req, res) => {
  const answers = Array.isArray(req.body?.answers) ? req.body.answers : [];
  if (answers.length === 0) return res.status(400).json({ error: "answers required" });

  // load correct keys
  const ids = answers.map(a => a.id);
  const { rows } = await pool.query(
    "SELECT id, correct_index, rationale FROM quiz_items WHERE id = ANY($1::int[])",
    [ids]
  );

  const byId = new Map(rows.map(r => [r.id, r]));
  let correct = 0;
  const detail = answers.map(a => {
    const key = byId.get(a.id);
    const isCorrect = key ? key.correct_index === a.choiceIndex : false;
    if (isCorrect) correct++;
    return { id: a.id, correct: isCorrect, rationale: key?.rationale || null };
  });

  res.json({ total: answers.length, correct, detail });
});

export default router;

E) Make sure your DB has the quiz table/rationale columns

Create db/setup_quiz.sql and paste:

CREATE TABLE IF NOT EXISTS quiz_items (
  id SERIAL PRIMARY KEY,
  subject_code VARCHAR(20) NOT NULL,
  topic_name TEXT,
  question TEXT NOT NULL,
  choices JSONB NOT NULL,
  correct_index INT NOT NULL CHECK (correct_index BETWEEN 0 AND 3),
  difficulty TEXT DEFAULT 'medium',
  tags TEXT[] DEFAULT '{}',
  rationale TEXT,
  source_note TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS quiz_items_code_idx ON quiz_items(subject_code);
CREATE INDEX IF NOT EXISTS quiz_items_tags_gin ON quiz_items USING GIN(tags);


Run it from your existing runner (if you have db/run-sql.js) or Railway:

node db/run-sql.js db/setup_quiz.sql
