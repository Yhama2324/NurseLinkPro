Add tables (SQL). If you use Prisma, translate to schema accordingly.

-- users table assumed (id uuid pk, tier text 'FREE'|'PRO'|'ELITE')

create table weekly_challenge_progress (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  week_start date not null,                      -- Monday of that ISO week
  answered int not null default 0,               -- total answered this week
  correct int not null default 0,
  streak_weeks int not null default 0,           -- consecutive 100/100 completions
  last_updated timestamptz not null default now(),
  unique (user_id, week_start)
);

create table weekly_challenge_events (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  week_start date not null,
  question_id text not null,
  correct boolean not null,
  latency_ms int,
  created_at timestamptz not null default now()
);

create table weekly_leaderboard (
  id uuid primary key default gen_random_uuid(),
  week_start date not null,
  user_id uuid not null references users(id) on delete cascade,
  score int not null default 0,
  rank int,
  created_at timestamptz not null default now(),
  unique (week_start, user_id)
);

3) Server helpers

Create server/utils/week.ts:

// server/utils/week.ts
export const startOfWeek = (d = new Date(), weekStartsOn = 1) => {
  const date = new Date(Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate()));
  const day = date.getUTCDay();
  const diff = (day < weekStartsOn ? 7 : 0) + day - weekStartsOn;
  date.setUTCDate(date.getUTCDate() - diff);
  date.setUTCHours(0,0,0,0);
  return date; // Monday 00:00:00 UTC if weekStartsOn=1
};

4) API routes (Express)

Create server/routes/weeklyChallenge.ts:

import { Router } from "express";
import { WEEKLY_CHALLENGE } from "../../src/config/challenge";
import { startOfWeek } from "../utils/week";
import pg from "pg";
const router = Router();
const pool = new pg.Pool({ connectionString: process.env.DATABASE_URL });

// calc score for leaderboard
function calcScore({ correct, events }:{correct:number, events:{correct:boolean, latency_ms:number|null}[]}) {
  const base = correct * 2;
  const speedBonus = Math.min(
    WEEKLY_CHALLENGE.points.speedBonusMax,
    Math.round(events.filter(e=>e.correct && (e.latency_ms ?? 9999) < 15000).length / 10)
  );
  return base + speedBonus;
}

// GET current progress
router.get("/me", async (req:any, res) => {
  const userId = req.user.id; // assume auth middleware
  const weekStart = startOfWeek(new Date(), WEEKLY_CHALLENGE.weekStartsOn);
  const { rows } = await pool.query(
    `select * from weekly_challenge_progress where user_id=$1 and week_start=$2`,
    [userId, weekStart]
  );
  const p = rows[0] ?? { answered:0, correct:0, streak_weeks:0 };
  res.json({ goal: WEEKLY_CHALLENGE.weeklyGoal, ...p });
});

// POST answer record (called after each MCQ submission)
router.post("/answer", async (req:any, res) => {
  const userId = req.user.id;
  const { questionId, correct, latencyMs } = req.body;
  const weekStart = startOfWeek(new Date(), WEEKLY_CHALLENGE.weekStartsOn);

  await pool.query("begin");
  await pool.query(
    `insert into weekly_challenge_events (user_id, week_start, question_id, correct, latency_ms)
     values ($1,$2,$3,$4,$5)`,
    [userId, weekStart, questionId, !!correct, latencyMs ?? null]
  );
  await pool.query(
    `insert into weekly_challenge_progress (user_id, week_start, answered, correct)
     values ($1,$2,1,$3) 
     on conflict (user_id, week_start) do update
       set answered = weekly_challenge_progress.answered + 1,
           correct  = weekly_challenge_progress.correct  + excluded.correct,
           last_updated = now()`,
    [userId, weekStart, correct ? 1 : 0]
  );
  await pool.query("commit");
  res.json({ ok:true });
});

// POST finalize week (award badge if 100/100)
router.post("/finalize", async (req:any, res) => {
  const userId = req.user.id;
  const weekStart = startOfWeek(new Date(), WEEKLY_CHALLENGE.weekStartsOn);

  const { rows: pr } = await pool.query(
    `select * from weekly_challenge_progress where user_id=$1 and week_start=$2`,
    [userId, weekStart]
  );
  const progress = pr[0];
  if (!progress) return res.status(400).json({ error:"No progress" });

  const { rows: ev } = await pool.query(
    `select correct, latency_ms from weekly_challenge_events where user_id=$1 and week_start=$2`,
    [userId, weekStart]
  );
  const score = calcScore({ correct: progress.correct, events: ev });

  // streak update
  let streak = progress.streak_weeks;
  if (progress.answered >= WEEKLY_CHALLENGE.weeklyGoal) {
    const { rows: prev } = await pool.query(
      `select streak_weeks from weekly_challenge_progress
       where user_id=$1 and week_start = ($2::date - interval '7 days')`,
      [userId, weekStart]
    );
    const prevStreak = prev[0]?.streak_weeks ?? 0;
    streak = prevStreak + 1;
    await pool.query(
      `update weekly_challenge_progress set streak_weeks=$1 where user_id=$2 and week_start=$3`,
      [streak, userId, weekStart]
    );
  }

  // upsert leaderboard
  await pool.query(
    `insert into weekly_leaderboard (week_start, user_id, score)
     values ($1,$2,$3)
     on conflict (week_start,user_id) do update set score=excluded.score`,
    [weekStart, userId, score]
  );

  res.json({
    completed: progress.answered >= WEEKLY_CHALLENGE.weeklyGoal,
    score,
    streakWeeks: streak
  });
});

export default router;

5) Cron to rank weekly

Run Sundays 23:59 (server time). Use node-cron or your platform’s scheduler.

// server/jobs/weeklyRank.ts
import pg from "pg";
import { startOfWeek } from "../utils/week";
const pool = new pg.Pool({ connectionString: process.env.DATABASE_URL });

export async function computeWeeklyRanks() {
  const now = new Date();
  const weekStart = startOfWeek(now);
  const { rows } = await pool.query(
    `select id, user_id, score from weekly_leaderboard where week_start=$1 order by score desc`,
    [weekStart]
  );
  let rank = 1;
  for (const r of rows) {
    await pool.query(`update weekly_leaderboard set rank=$1 where id=$2`, [rank++, r.id]);
  }
}
